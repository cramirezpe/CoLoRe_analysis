21""" 

22===================================================== 

23visufunc.py : Healpix visualization functions 

24===================================================== 

25 

26This module provides functions to display Healpix map. 

27 

28Map projections 

29--------------- 

30 

31- :func:`mollview` displays a map using Mollweide projection (full sky) 

32- :func:`gnomview` displays a map using Gnomonic projection (local map) 

33- :func:`cartview` displays a map using Cartesian projection 

34- :func:`orthview` displays a map using Orthographic projection (full or half sky) 

35- :func:`azeqview` displays a map using Azimuthal equidistant projection 

36 

37Graticules 

38---------- 

39 

40- :func:`graticule` adds a graticule to the current map 

41- :func:`delgraticules` deletes all graticules of a map 

42 

43Tracing lines or points 

44----------------------- 

45 

46- :func:`projplot` plots data points on the current map 

47- :func:`projscatter` displays scatter points 

48- :func:`projtext` display a text on the current map 

49"""

108 If None, will display a blank map, useful for overplotting. 

109 fig : int or None, optional

115 of angle *psi* around this direction is applied. 

116 coord : sequence of character, optional 

117 Either one of 'G', 'E' or 'C' to describe the coordinate 

118 system of the map, or a sequence of 2 of these to rotate

120 unit : str, optional

122 xsize : int, optional 

123 The size of the image. Default: 800 

124 title : str, optional 

125 The title of the plot. Default: 'Mollweide view' 

126 nest : bool, optional 

127 If True, ordering scheme is NESTED. Default: False (RING) 

128 min : float, optional 

129 The minimum range value 

130 max : float, optional 

131 The maximum range value 

132 flip : {'astro', 'geo'}, optional 

133 Defines the convention of projection : 'astro' (default, east towards left, west towards right) 

134 or 'geo' (east towards right, west towards left) 

135 remove_dip : bool, optional 

136 If :const:`True`, remove the dipole+monopole 

137 remove_mono : bool, optional

152 log= logarithmic color mapping, default: None (linear color mapping) 

153 cmap : a color map 

154 The colormap to use (see matplotlib.cm) 

155 badcolor : str 

156 Color to use to plot bad values 

157 bgcolor : str 

158 Color to use for background 

159 hold : bool, optional 

160 If True, replace the current Axes by a MollweideAxes. 

161 use this if you want to have multiple maps on the same 

162 figure. Default: False 

163 sub : int, scalar or sequence, optional 

164 Use only a zone of the current figure (same syntax as subplot). 

165 Default: None

178 # Create the figure

181 # Ensure that the nside is valid

183 pixelfunc.check_nside(nside, nest=nest)

187 extent = (0.02, 0.05, 0.96, 0.9)

189 f = pylab.gcf() 

190 left, bottom, right, top = np.array(f.gca().get_position()).ravel() 

191 extent = (left, bottom, right - left, top - bottom) 

192 f.delaxes(f.gca()) 

193 else: # using subplot syntax 

194 f = pylab.gcf() 

195 if hasattr(sub, "__len__"): 

196 nrows, ncols, idx = sub 

197 else: 

198 nrows, ncols, idx = sub // 100, (sub % 100) // 10, (sub % 10)

201 c, r = (idx - 1) % ncols, (idx - 1) // ncols 

202 if not margins: 

203 margins = (0.01, 0.0, 0.0, 0.02) 

204 extent = ( 

205 c * 1.0 / ncols + margins[0], 

206 1.0 - (r + 1) * 1.0 / nrows + margins[1], 

207 1.0 / ncols - margins[2] - margins[0], 

208 1.0 / nrows - margins[3] - margins[1], 

209 ) 

210 extent = ( 

211 extent[0] + margins[0], 

212 extent[1] + margins[1], 

213 extent[2] - margins[2] - margins[0], 

214 extent[3] - margins[3] - margins[1], 

215 )

217 # f=pylab.figure(fig,figsize=(8.5,5.4))

219 # Starting to draw : turn interactive off 

220 wasinteractive = pylab.isinteractive() 

221 pylab.ioff() 

222 try: 

223 if map is None: 

224 map = np.zeros(12) + np.inf 

225 cbar = False 

226 map = pixelfunc.ma_to_array(map) 

227 ax = PA.HpxMollweideAxes( 

228 f, extent, coord=coord, rot=rot, format=format2, flipconv=flip 

229 ) 

230 f.add_axes(ax) 

231 if remove_dip: 

232 map = pixelfunc.remove_dipole( 

233 map, gal_cut=gal_cut, nest=nest, copy=True, verbose=True 

234 ) 

235 elif remove_mono: 

236 map = pixelfunc.remove_monopole( 

237 map, gal_cut=gal_cut, nest=nest, copy=True, verbose=True 

238 ) 

239 img = ax.projmap( 

240 map, 

241 nest=nest, 

242 xsize=xsize, 

243 coord=coord, 

244 vmin=min, 

245 vmax=max, 

246 cmap=cmap, 

247 badcolor=badcolor, 

248 bgcolor=bgcolor, 

249 norm=norm, 

250 ) 

251 if cbar: 

252 im = ax.get_images()[0] 

253 b = im.norm.inverse(np.linspace(0, 1, im.cmap.N + 1)) 

254 v = np.linspace(im.norm.vmin, im.norm.vmax, im.cmap.N) 

255 if matplotlib.__version__ >= "0.91.0": 

256 cb = f.colorbar( 

257 im, 

258 ax=ax, 

259 orientation="horizontal", 

260 shrink=0.5, 

261 aspect=25, 

262 ticks=PA.BoundaryLocator(nlocs, norm), 

263 pad=0.05, 

264 fraction=0.1, 

265 boundaries=b, 

266 values=v, 

267 format=format, 

268 ) 

269 else: 

270 # for older matplotlib versions, no ax kwarg 

271 cb = f.colorbar( 

272 im, 

273 orientation="horizontal", 

274 shrink=0.5, 

275 aspect=25, 

276 ticks=PA.BoundaryLocator(nlocs, norm), 

277 pad=0.05, 

278 fraction=0.1, 

279 boundaries=b, 

280 values=v, 

281 format=format, 

282 ) 

283 cb.solids.set_rasterized(True) 

284 ax.set_title(title) 

285 if not notext: 

286 ax.text( 

287 0.86, 

288 0.05, 

289 ax.proj.coordsysstr, 

290 fontsize=14, 

291 fontweight="bold", 

292 transform=ax.transAxes, 

293 ) 

294 if cbar: 

295 cb.ax.text( 

296 0.5, 

297 -1.0, 

298 unit, 

299 fontsize=14, 

300 transform=cb.ax.transAxes, 

301 ha="center", 

302 va="center", 

303 ) 

304 f.sca(ax) 

305 finally: 

306 pylab.draw() 

307 if wasinteractive: 

308 pylab.ion() 

309 # pylab.show() 

310 if return_projected_map: 

311 return img
